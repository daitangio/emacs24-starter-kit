;; -*- coding: utf-8-emacs -*-
(message "MacOSX Anakin...")
;; Fix a pair of quirks on macosx keyboard. 
;; Ref http://xahlee.org/emacs/keyboard_shortcuts.html
;; Benefit: Alt-x is equal to M-x so Alt-x enable minibuffer command mode 
(global-set-key (kbd "M-à") "#" )
(global-set-key (kbd "M-ò") "@" )
(global-set-key (kbd "M-è") "[" )
;; Alt-shift on mac for { 
(global-set-key (kbd "M-é") "{" )
;; M-* is normally bound to pop-tag-mark, we remap to }
(global-set-key (kbd "M-*") "}" )
(global-set-key (kbd "M-+") "]" )
;; Optional but useful:
(global-set-key (kbd "M-e") "€" )
;;; Fixer  (global-unset-key (kbd "C-_"))

;; Remap common unused keys lik ° and §
(global-set-key (kbd "°") (kbd "`"))
(global-set-key (kbd "§") (kbd "~"))
